                    SOURCE_DIR          "/home/mundus/base/esp/esp32_cam_surv_ai/server/analysis/inference/build/xnnpack"
                    BINARY_DIR          "/home/mundus/base/esp/esp32_cam_surv_ai/server/analysis/inference/build/_deps/xnnpack-build"
                    CONFIGURE_COMMAND   ""
